<!-- Script para cambiar el color del navbar al desplazarse -->
<script>
    window.addEventListener('scroll', function() {
        const navbar = document.getElementById('navbar');
        const navbarLinks = document.querySelectorAll('#navbar-links a');
        const navbarLogo = document.getElementById('navbar-logo');
        const navbarButton = document.querySelector('#navbar-button a');

        if (window.scrollY > 50) {
            navbar.classList.add('bg-green-600', 'shadow-xl');
            navbar.classList.remove('bg-white');

            navbarLogo.classList.add('text-white');
            navbarLogo.classList.remove('text-green-600');

            navbarLinks.forEach(link => {
                link.classList.add('text-white');
                link.classList.remove('text-gray-700');
                link.classList.add('hover:text-gray-200');
                link.classList.remove('hover:text-green-600');
            });

            navbarButton.classList.add('bg-white', 'text-green-600');
            navbarButton.classList.remove('bg-green-600', 'text-white');
        } else {
            navbar.classList.add('bg-white');
            navbar.classList.remove('bg-green-600', 'shadow-xl');

            navbarLogo.classList.add('text-green-600');
            navbarLogo.classList.remove('text-white');

            navbarLinks.forEach(link => {
                link.classList.add('text-gray-700');
                link.classList.remove('text-white');
                link.classList.add('hover:text-green-600');
                link.classList.remove('hover:text-gray-200');
            });

            navbarButton.classList.add('bg-green-600', 'text-white');
            navbarButton.classList.remove('bg-white', 'text-green-600');
        }
    });
</script>
